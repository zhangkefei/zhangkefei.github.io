发表于 2014-09-22 更新于 2021-05-28 分类于 mysql

mysql数据库常用命令

  • show databases;
  • use databaseName;
  • show tables;
  • show columns from tableName;
  • show table status like tableName;
  • create database databaseName;
  • 'create a user'
    • 'For more information,please look this.'
  • create table if not exists tableName (id int unsigned auto_increment,title varchar(100) not null,name char(10) not null default '',date date comment 'the correct date',primary key(id))engine=InnoDB default charset=utf8;
  • drop table tableName;
  • insert into tableName (id,title,date) values ('1','title',now()),('1','title',now());
  • select * from tableName;
    • select column1,column2 from tableName [where condition1 [and][or] condition2][offset M][limit n];
    • select column1 from tableName where column1 like '%com%';
    • select column1,column2,columns from tableName order by column1 [asc [desc]];
  • update tableName set field1=new-value,field2=new-value [where conditions]
  • delete from tableName [where conditions]
  • 'alter table'
    • alter table tableName drop column1;
    • alter table tableName add column1 int not null default '0';
    • alter table tableName add column1 int first;
    • alter table tableName add column1 int after column2;
    • alter table tableName modify column1 int;
    • alter table tableName change column1 column2 int not null;
    • atter table tableName rename to tableName2;
    • alter table tableName add primary key (columnName);
  • get create mysql sentence
    • show create table table_name
  • copy a database

    • mysql -u root -p123456

    • mysql>CREATE DATABASE newdb DEFAULT CHARACTER SET UTF8 COLLATE UTF8_GENERAL_CI;

    • mysqldump db1 -u root -p123456 --add-drop-table | mysql newdb -u root -p123456

  • set table auto_increment value
    • ALTER TABLE AUTO_INCREMENT = 100;
  • 查看MYSQL数据库占用空间大小
    1. use information_schema;
    2. select concat(round(sum(DATA_LENGTH)/1024/1024+sum(INDEX_LENGTH)/1024/1024),'M') from information_schema.tables where table_schema='wavelab';
  • MySQL 表字段唯一性约束设置方法unique
    • 建表时加上唯一性约束
    • 给已经建好的表加上唯一性约束
